-- ==========================================
-- 1. BASE TABLES (Foods & Logs)
-- ==========================================
CREATE TABLE IF NOT EXISTS foods (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  name TEXT NOT NULL,
  brand TEXT,
  serving_size NUMERIC NOT NULL,
  serving_unit TEXT NOT NULL,
  calories NUMERIC NOT NULL,
  protein NUMERIC NOT NULL,
  carbs NUMERIC NOT NULL,
  fat NUMERIC NOT NULL,
  created_by UUID REFERENCES auth.users(id) 
);

CREATE TABLE IF NOT EXISTS logs (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES auth.users(id) NOT NULL DEFAULT auth.uid(),
  food_id UUID REFERENCES foods(id) NOT NULL,
  date DATE NOT NULL DEFAULT CURRENT_DATE,
  meal_time TEXT CHECK (meal_time IN ('Breakfast', 'Lunch', 'Dinner', 'Snack', 'Other')),
  servings_consumed NUMERIC NOT NULL,
  created_at TIMESTAMPTZ DEFAULT NOW()
);

-- ==========================================
-- 2. AUTOMATED VIEWS (Macro Calculators)
-- ==========================================
CREATE OR REPLACE VIEW daily_summary AS
SELECT 
  l.user_id,
  l.date,
  ROUND(SUM(f.calories * l.servings_consumed), 1) AS total_calories,
  ROUND(SUM(f.protein * l.servings_consumed), 1) AS total_protein,
  ROUND(SUM(f.carbs * l.servings_consumed), 1) AS total_carbs,
  ROUND(SUM(f.fat * l.servings_consumed), 1) AS total_fat
FROM logs l
JOIN foods f ON l.food_id = f.id
GROUP BY l.user_id, l.date;

CREATE OR REPLACE VIEW meal_time_summary AS
SELECT 
  l.user_id,
  l.date,
  l.meal_time,
  ROUND(SUM(f.calories * l.servings_consumed), 1) AS calories,
  ROUND(SUM(f.protein * l.servings_consumed), 1) AS protein,
  ROUND(SUM(f.carbs * l.servings_consumed), 1) AS carbs,
  ROUND(SUM(f.fat * l.servings_consumed), 1) AS fat
FROM logs l
JOIN foods f ON l.food_id = f.id
GROUP BY l.user_id, l.date, l.meal_time;

-- ==========================================
-- 3. ADVANCED TABLES (Goals, Meals, Recipes)
-- ==========================================
CREATE TABLE IF NOT EXISTS user_goals (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES auth.users(id) NOT NULL DEFAULT auth.uid(),
  date DATE NOT NULL DEFAULT CURRENT_DATE,
  target_calories NUMERIC,
  target_protein NUMERIC,
  target_carbs NUMERIC,
  target_fat NUMERIC,
  UNIQUE(user_id, date) 
);

CREATE TABLE IF NOT EXISTS meals (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES auth.users(id) NOT NULL DEFAULT auth.uid(),
  name TEXT NOT NULL
);

CREATE TABLE IF NOT EXISTS meal_items (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  meal_id UUID REFERENCES meals(id) ON DELETE CASCADE,
  food_id UUID REFERENCES foods(id) NOT NULL,
  servings NUMERIC NOT NULL
);

CREATE TABLE IF NOT EXISTS recipes (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  user_id UUID REFERENCES auth.users(id) NOT NULL DEFAULT auth.uid(),
  name TEXT NOT NULL,
  total_servings NUMERIC NOT NULL, 
  generated_food_id UUID REFERENCES foods(id) 
);

CREATE TABLE IF NOT EXISTS recipe_ingredients (
  id UUID PRIMARY KEY DEFAULT uuid_generate_v4(),
  recipe_id UUID REFERENCES recipes(id) ON DELETE CASCADE,
  food_id UUID REFERENCES foods(id) NOT NULL,
  servings NUMERIC NOT NULL
);

-- ==========================================
-- 4. RPC: EXPLODE RECIPE INTO LOGS
-- ==========================================
CREATE OR REPLACE FUNCTION log_exploded_recipe(
  p_recipe_id UUID,
  p_servings_consumed NUMERIC, 
  p_date DATE,
  p_meal_time TEXT
) RETURNS VOID AS $$
DECLARE
  v_recipe_total_servings NUMERIC;
  v_multiplier NUMERIC;
BEGIN
  SELECT total_servings INTO v_recipe_total_servings 
  FROM recipes WHERE id = p_recipe_id AND user_id = auth.uid();

  v_multiplier := p_servings_consumed / v_recipe_total_servings;

  INSERT INTO logs (user_id, food_id, date, meal_time, servings_consumed)
  SELECT auth.uid(), food_id, p_date, p_meal_time, servings * v_multiplier
  FROM recipe_ingredients WHERE recipe_id = p_recipe_id;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

-- ==========================================
-- 5. TRIGGER: NEW USER DEFAULT GOALS
-- ==========================================
CREATE OR REPLACE FUNCTION public.handle_new_user()
RETURNS trigger AS $$
BEGIN
  INSERT INTO public.user_goals (user_id, date, target_calories, target_protein, target_carbs, target_fat)
  VALUES (new.id, CURRENT_DATE, 2000, 150, 200, 65);
  RETURN new;
END;
$$ LANGUAGE plpgsql SECURITY DEFINER;

DROP TRIGGER IF EXISTS on_auth_user_created ON auth.users;
CREATE TRIGGER on_auth_user_created
  AFTER INSERT ON auth.users
  FOR EACH ROW EXECUTE PROCEDURE public.handle_new_user();

-- ==========================================
-- 6. ROW LEVEL SECURITY (RLS) POLICIES
-- ==========================================
ALTER TABLE foods ENABLE ROW LEVEL SECURITY;
ALTER TABLE logs ENABLE ROW LEVEL SECURITY;
ALTER TABLE user_goals ENABLE ROW LEVEL SECURITY;
ALTER TABLE meals ENABLE ROW LEVEL SECURITY;
ALTER TABLE meal_items ENABLE ROW LEVEL SECURITY;
ALTER TABLE recipes ENABLE ROW LEVEL SECURITY;
ALTER TABLE recipe_ingredients ENABLE ROW LEVEL SECURITY;

CREATE POLICY "Foods viewable by all" ON foods FOR SELECT USING (true);
CREATE POLICY "Users can create foods" ON foods FOR INSERT TO authenticated WITH CHECK (true);

CREATE POLICY "Manage own logs" ON logs FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "Manage own goals" ON user_goals FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "Manage own meals" ON meals FOR ALL USING (auth.uid() = user_id);
CREATE POLICY "Manage own recipes" ON recipes FOR ALL USING (auth.uid() = user_id);

CREATE POLICY "Manage own meal items" ON meal_items FOR ALL USING (
  EXISTS (SELECT 1 FROM meals WHERE id = meal_items.meal_id AND user_id = auth.uid())
);
CREATE POLICY "Manage own recipe ingredients" ON recipe_ingredients FOR ALL USING (
  EXISTS (SELECT 1 FROM recipes WHERE id = recipe_ingredients.recipe_id AND user_id = auth.uid())
);
